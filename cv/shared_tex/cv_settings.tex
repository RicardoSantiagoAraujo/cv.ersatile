% Create scores in a range
\newcommand{\score}[1]{%
  \newcommand{\bulletGap}{\hspace{0.4em}}%
  \foreach \i in {1, ..., 5} {%
    \ifnum \the\numexpr \i - 1  < #1% subtract 1 to act as "<="
        \textcolor{myIconColor}{\bullet\bulletGap}% Filled dot for score
    \else%
        \textcolor{myIconColor}{\circ\bulletGap}% Open dot for no score
    \fi%
  }%
}
