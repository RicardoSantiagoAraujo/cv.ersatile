

\tcbset{
       myPictureStyleDefault/.style={
            % breakable,
              left=0pt,
              right=0pt,
              top=0pt,
              bottom=0pt,
              colback=white,
              colframe=myGrayMed,
              enlarge left by=0mm,
              boxsep=0pt,
              boxrule=1pt,
              arc=0pt,
              % outer arc=3pt,
              % title=\imagename,
              finish={
                     \isDraft{
                            \node[fill=myGrayLight!50!white,fill opacity=0.5,text opacity=1]
                            at (frame.center) {\scriptsize\bfseries\ttfamily\color{myGrayDark!50!black}\imagename};}{}
              }
       },
       % myPictureStyleExtras/.style={}
}

\NewDocumentCommand{\myStyledGraphic}
{
       m % graphics width
       O{\myPictureCropLeft}
       O{\myPictureCropBottom}
       O{\myPictureCropRight}
       O{\myPictureCropTop}
       m % graphics file
       O{}% Extra styles to overwrite default if desired
}
{%
       \tcbincludegraphics % tcbincludegraphics is part of tcolorbox package
       [%
              width=#1,
              myPictureStyleDefault,
              #7,
       %      includegraphics options:
              graphics options={%
                   width=3cm,
                   trim= #2 #3 #4 #5  % trim={left bottom right top},clip]
                   }%
       ]%
       {#6}%
}
