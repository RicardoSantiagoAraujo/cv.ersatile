\newcommand{\applyFonts}[2]{
    % #1 first option
    % #2 backup
    \IfFontExistsTF{#1}
    {%
        \setmainfont{#1}%
        %[BoldFont=CambriaB]% shortcut fix to use bold cambria on windows
    }%
    {% Fallback option
        \setmainfont{#2}%
    }%
}
